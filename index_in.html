in演算子の使用方法

      
        // in演算子の使用方法

        // （１）オブジェクトにインデックスがあるか確認する
        let cats = {
          27: "mike",
          104: "kuro",
          256: "tora",
          home: true,
          food: "カリカリ"
        };

        if (256 in cats) {
          console.log("toraは居る"); // toraは居る
        } else {
          console.log("toraは居ない");
        }

        if (50 in cats) {
          console.log("No.50は居る");
        } else {
          console.log("No.50は居ない"); // No.50は居ない
        }

        // （２）オブジェクトにプロパティがあるか確認する
        if ("home" in cats) {
          console.log("homeはある");
        } else {
          console.log("homeはない");
        }

        // ※実践的には↓
        if (cats.food) {
          console.log(cats.food); // カリカリ
        }

        // （３）forと組み合わせて、オブジェクトのプロパティをループ処理する
        for (let p in cats) {
          console.log(p);
        }
        // 27
        // 104
        // 256
        // home
        // food

in演算子とhasOwnPropertyの差

      
        // in はプロトタイプチェーンを辿って走査
        // hasOwnPropertyは辿らない（自分自身が持っているもののみ）

        const myCats = {
          white: "シロ",
          black: "クロ",
          mike: "ミケ"
        };

        // inの場合
        console.log("'white' in myCats: ", "white" in myCats); // 'white' in myCats:  true
        console.log("'toString' in myCats: ", "toString" in myCats); // 'toString' in myCats:  true

        // hasOwnProperty
        console.log("myCats.hasOwnProperty('white'): ", myCats.hasOwnProperty("white")); // myCats.hasOwnProperty('white'):  true
        console.log("myCats.hasOwnProperty('toString'): ", myCats.hasOwnProperty("toString")); // myCats.hasOwnProperty('toString'):  false
