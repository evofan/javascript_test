Array.prototype.find()

テスト内容はChromeのコンソールに表示＋参考ページを元にサンプル作成

find()を使った例

      
        const cat_array1 = [
          { name: "mike", age: 10, sex: "female" },
          { name: "kuro", age: 6, sex: "male" },
          { name: "tora", age: 3, sex: "male" }
        ];
        let targetAge = 6;
        let targetCat = cat_array1.find(el => el.age === 6);
        console.log(targetCat); // {name: "kuro", age: 6}

        // find()以外の「探す」系メソッド

        // □ filter() ... 全部探す
        let targetCat2 = cat_array1.filter(el => el.sex === "male");
        console.log(targetCat2); // [ {name: "kuro", age: 6, sex: "male"}, {name: "tora", age: 3, sex: "male"} ]

        // □ findIndex() ... インデックスを探す
        let targetCat3 = cat_array1.findIndex(el => el.name === "tora");
        console.log(targetCat3); // 2
        console.log(cat_array1[targetCat3]); // {name: "tora", age: 3, sex: "male"}

        // □ some() ... あるか（1つ以上存在するか）を調べる
        let targetCat4 = cat_array1.some(el => el.age >= 10);
        console.log(targetCat4); // true
