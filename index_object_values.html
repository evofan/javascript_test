Object.values()

テスト内容はChromeのConsoleに表示＋参考ページを元にサンプル作成

Object.{keys, values, entries}

      

        let obj_cat = {
          mike: { color: "三毛", age: 5 },
          tama: { color: "キジ", age: 3 },
          kuro: { color: "黒", age: 1 }
        };
        console.log(Object.values(obj_cat)); // [ { color: "三毛", age: 5 }, { color: "キジ", age: 3 }, { color: "黒", age: 1 } ]
        console.log(Object.keys(obj_cat)); // [ "mike", "tama", "kuro" ]
        console.log(Object.entries(obj_cat)); // [ [ "mike", { color: "三毛", age: 5 } ], [ "tama", { color: "キジ", age: 3 } ], [ "kuro", { color: "黒", age: 1 } ] ]
